window.calculateHand = function() { const winIdx = parseInt(document.getElementById('winner_idx').value); let handPoints = [0, 0, 0, 0]; // 1. Calculate Base Penalties const winEndingTwos = parseInt(document.getElementById(`p${winIdx}_val`).value) || 0; const winTwosBonus = winEndingTwos * 5; for(let i=0; i<4; i++) { if(i !== winIdx) { let cards = parseInt(document.getElementById(`p${i}_val`).value) || 0; let heldTwos = parseInt(document.getElementById(`p${i}_twos`).value) || 0; // Core point calculation (Cards + 5 per Held 2) let subtotal = cards + (heldTwos * 5); // Apply Multipliers based on card count if (cards === 13) { subtotal = subtotal * 3; // TRIPLE for 13 cards } else if (cards >= 10) { subtotal = subtotal * 2; // DOUBLE for 10-12 cards } // Add the Ante (5) + Winner's Ending 2s bonus (5 per 2) let totalPenalty = subtotal + 5 + winTwosBonus; handPoints[i] -= totalPenalty; handPoints[winIdx] += totalPenalty; } } // 2. Card Differences (Loser vs Loser side-bets) let cardsArr = [0,0,0,0]; for(let i=0; i<4; i++) { if(i !== winIdx) cardsArr[i] = parseInt(document.getElementById(`p${i}_val`).value) || 0; } for(let i=0; i<4; i++) { for(let j=0; j<4; j++) { if(cardsArr[i] > cardsArr[j]) { let diff = cardsArr[i] - cardsArr[j]; handPoints[i] -= diff; handPoints[j] += diff; } } } // 3. Specials Distribution for(let i=0; i<4; i++) { let specVal = parseInt(document.getElementById(`p${i}_special`).value) || 0; if(specVal > 0) { for(let j=0; j<4; j++) { if(i !== j) { handPoints[j] -= specVal; handPoints[i] += specVal; } } } } if(!sessionData.history) sessionData.history = []; sessionData.history.push(handPoints); handPoints.forEach((pts, i) => sessionData.totals[i] += pts); syncToFirebase(); };
